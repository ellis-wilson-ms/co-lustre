Increase DNE update llog chunk size from 8KB to 32KB, so
one cross-MDT operation can includes more update records,
and it can create striped directory with more stripe count.

(2 comments)

It looks like the problem is from trying to read an existing header that has a smaller record size but also has some file data:

    llog_osd_read_header(): error reading log header from [0x5c000040f:0x1:0x0]: rc = -14

The -EFAULT error is (with the patch applied) is explicitly set in llog_osd_read_header():

        lgi->lgi_buf.lb_buf = handle->lgh_hdr;
        lgi->lgi_buf.lb_len = handle->lgh_hdr_size;

static int llog_osd_read_header(const struct lu_env *env,
                                struct llog_handle *handle)
{
        :
        rc = dt_record_read(env, o, &lgi->lgi_buf, &lgi->lgi_off);
        if (rc) {
                CERROR("%s: error reading log header from "DFID": rc = %d\n",
                       o->do_lu.lo_dev->ld_obd->obd_name,
                       PFID(lu_object_fid(&o->do_lu)), rc);
                RETURN(rc);
        }

int dt_record_read(const struct lu_env *env, struct dt_object *dt,
                   struct lu_buf *buf, loff_t *pos)
{
        :
        size = dt->do_body_ops->dbo_read(env, dt, buf, pos);
        if (size < 0)
                return size; 
        return (size == (ssize_t)buf->lb_len) ? 0 : -EFAULT;
}

So the problem is that dt_record_read() is trying to read the full header size, and returns an error if e.g. the llog file is smaller than the specified header size (e.g. reading an old llog with 8KB header + 16KB of llog records < 32KB requested read).

Probably the best solution is to always start by trying to read the larger llog header size (which will succeed for new llog files, and old llog files that have more than a few blocks of data), and if -EFAULT is returned and hdr_size > LLOG_MIN_CHUNK_SIZE, then retry with hdr_size = LLOG_MIN_CHUNK_SIZE:


static int llog_osd_read_header(const struct lu_env *env,
                                struct llog_handle *handle)
{
        :
        lgi->lgi_buf.lb_buf = handle->lgh_hdr;
        lgi->lgi_buf.lb_len = handle->lgh_hdr_size;

        rc = dt_record_read(env, o, &lgi->lgi_buf, &lgi->lgi_off);
        if (rc) {
                CERROR("%s: error reading log header from "DFID": rc = %d\n",
                       o->do_lu.lo_dev->ld_obd->obd_name,
                       PFID(lu_object_fid(&o->do_lu)), rc);
                RETURN(rc);
        }
