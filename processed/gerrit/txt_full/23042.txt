rc = mgs_check(env, mgs, fsdb, mti, logname, mti->mti_svname,

static int mgs_check(const struct lu_env *env, struct mgs_device *mgs,
			struct fs_db *fsdb, struct mgs_target_info *mti,
			char *logname, char *devname, char *comment)
{
         struct cfg_marker *mc_marker;
         rc = llog_process(env, loghandle, mgs_check_record_match,
			(void *)mc_marker, NULL);
}

static int mgs_check_record_match(const struct lu_env *env,
				struct llog_handle *llh,
				struct llog_rec_hdr *rec, void *data)
{
struct cfg_marker *mc_marker = data;
struct cfg_marker *marker;
struct lustre_cfg *lcfg = REC_DATA(rec);
marker = lustre_cfg_buf(lcfg, 1);
  if ((strcmp(mc_marker->cm_comment, marker->cm_comment) == 0) &&
		(strcmp(mc_marker->cm_tgtname, marker->cm_tgtname) == 0) &&
		!(marker->cm_flags & CM_SKIP)) {
           rc = LLOG_MATCH_RECORD;
    }
   RETURN(rc);
}

It looks like the two options produce slightly different results - the first one prevents insertion of the duplicate entry into the log, while the second only returns an error during log processing if it hits a duplicate log.  It would be preferable to prevent the duplicate pool to be added to the log, rather than handling it after the fact.  That said, if there are sites that already have duplicate pool names in the logs, then this needs to be handled gracefully (i.e. not abort config log processing).

My original through was to have "lctl pool_new <arg>" do an internal call to "llapi_poollist(argv[1])" to see if the pool already exists, instead of complicating the kernel code more, but this won't work if the MGS is on a separate node.
