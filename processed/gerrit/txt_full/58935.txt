--- Commit Message ---
LU-18946 ptlrpc: make loopback enforcement optional

this can be useful if you plan to mount clients using different NIDs
within a single VM:

$ export MODOPTS_PTLRPC=use_loop=0
$ export MODOPTS_LNET="lnet_peer_discovery_disabled=1
	networks=tcp0(enp0s4),tcp1(enp0s5)"
$ bash llmount.sh
$ lctl list_nids
192.168.123.180@tcp
192.168.122.2@tcp1
$ export MGS="192.168.123.180@tcp,192.168.122.2@tcp"
$ mount -t lustre $MGS:/lustre /mnt/lustre
$ mount -t lustre network=tcp1 $MGS:/lustre /mnt/lustre2
$ lctl get_param -N mdt.*.exports.*
mdt.lustre-MDT0000.exports.192.168.122.2@tcp1
mdt.lustre-MDT0000.exports.192.168.123.180@tcp
mdt.lustre-MDT0000.exports.clear
