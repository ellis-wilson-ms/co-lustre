For kernel 3.10.x kref_put(x, NULL);  although semantics/syntax wise is correct, it generates a backtrace as shown in above link. Please note kref_put(x, NULL); works fine under 4.18.x kernels. 

Here is a very basic reproducer. Can be reproduced by just insmod.

struct foo {
	int x;
	struct kref kref_count;
};

static int simple_init(void)
{
	struct foo f;

	kref_init(&f.kref_count);
	printk(KERN_ERR "Val 0: %d\n", kref_read(&f.kref_count));

	kref_get(&f.kref_count);
	printk(KERN_ERR "Val 1: %d\n", kref_read(&f.kref_count));

	kref_put(&f.kref_count, NULL);
	printk(KERN_ERR "Val 3: %d\n", kref_read(&f.kref_count));

	return 0;
}

static void simple_cleanup(void)
{
}
module_init(simple_init);
module_exit(simple_cleanup);
