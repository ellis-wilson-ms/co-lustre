--- Description ---
For example,

[root@centos7-3 tests]# cp /etc/passwd /mnt/lustre/
[root@centos7-3 tests]# lfs path2fid /mnt/lustre/passwd 
[0x200000401:0x3:0x0]
[root@centos7-3 tests]# lfs fid2path /mnt/lustre 0x200000401:0x3:0x0 
/mnt/lustre/passwd
[root@centos7-3 tests]# lfs fid2path /mnt/lustre_sdfsf 0x200000401:0x3:0x0
/mnt/lustre_sdfsf/passwd
