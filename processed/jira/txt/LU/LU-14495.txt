#!/bin/bash

dirname=%s/%s
mkdir -p $dirname
nrdt=`lfs osts $dirname | awk '/-OST/{count += 1} END{print count}'`
nrmd=`lfs mdts $dirname | awk '/-MDT/{count += 1} END{print count}'`

dirname=${dirname}/$SLURM_JOB_ID
# striped directory, if supported
# This is a temporary hack to circumvent the
# problems described in INTL-201
c_count=$((RANDOM %% (nrmd + 1)))
s_offset=$((RANDOM %% nrmd))
# Setting offset is working, but creating directory
# directly utilising -D option still fails (see INTL-201)
lfs setdirstripe -c $c_count -i $s_offset $dirname > /dev/null 2>&1
lfs setdirstripe -c $c_count -i $s_offset -D $dirname > /dev/null 2>&1
# Temporary commented out till INTL-201 is landed and/or include into build
##lfs setdirstripe -c $((RANDOM %% (nrmd + 1))) -i $((RANDOM %% nrmd)) $dirname > /dev/null 2>&1
if [ ! -d $dirname ]; then
        # fallback to remote directory
        lfs setdirstripe -i $s_offset $dirname > /dev/null 2>&1
        if [ ! -d $dirname ]; then
                # fallback to striped directory
                lfs setdirstripe -c $c_count $dirname > /dev/null 2>&1
                if [ ! -d $dirname ]; then
                        # fallback to regular directory
                        mkdir $dirname
                fi
        fi
fi
