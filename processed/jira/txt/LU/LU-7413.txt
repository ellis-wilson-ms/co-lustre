        for (i = 0; i if (rdbuf->rb_bufs[i] != NULL) {
                        OBD_FREE(rdbuf->rb_bufs[i]->lb_buf,
                                 rdbuf->rb_bufs[i]->lb_len);
                        OBD_FREE_PTR(rdbuf->rb_bufs[i]);
                }
	}
