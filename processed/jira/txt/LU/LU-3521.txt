--- Description ---

[global]
directory=/mnt/lustre
filename=ssf
size=1m
blocksize=1m
direct=0
ioengine=sync
numjobs=32
group_reporting=1
fallocate=none
runtime=60
stonewall

$ git describe
v2_4_50_0

$ sh autogen.sh && ./configure --enable-invariants && make
$ sudo FSTYPE=zfs sh ./lustre/tests/llmount.sh
$ sudo fio /tmp/test.fio
