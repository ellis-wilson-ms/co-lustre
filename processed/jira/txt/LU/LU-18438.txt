                if (conflict != NULL) {
                  rc = osc_extent_wait(env, conflict, OES_INV);  this assumes waiting for completion of RPC in progress
                }

    cl_io_end
    cl_io_unlock
      osc_lock_cancel
        osc_lock_detach.
