Here is a testcase

sleep 1

flock -e /mnt/lustre/ff -c 'sleep 5' &

sleep 1

echo "umount -f /mnt/mds1"

umount -f /mnt/mds1 || true

killall flock

ps ax

echo "umount -f /mnt/lustre"

umount -f /mnt/lustre || true

cleanupall -f || error "cleanup failed"
