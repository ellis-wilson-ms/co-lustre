In revalidate_statahead_dentry() if md_revalidate_lock() succeeds but the dentry is stale then the lock reference is leaked:

                rc = md_revalidate_lock(ll_i2mdexp(dir), &it,
                                        ll_inode2fid(inode), &bits);
                if (rc == 1) {
                        if ((*dentryp)->d_inode == NULL) {
                                struct dentry *alias;

                                alias = ll_splice_alias(inode, *dentryp);
                                if (IS_ERR(alias))
                                        GOTO(out, rc = PTR_ERR(alias));
                                *dentryp = alias;
                                /* statahead prepared this inode, transfer inode          
                                 * refcount from sa_entry to dentry */
                                entry->se_inode = NULL;
                        } else if ((*dentryp)->d_inode != inode) {
                                /* revalidate, but inode is recreated */
                                CERROR("%s: stale dentry %.*s inode "
                                       DFID", statahead inode "DFID"\n",
                                        ll_get_fsname((*dentryp)->d_inode->i_sb,
                                                      NULL, 0),
                                        (*dentryp)->d_name.len,
                                        (*dentryp)->d_name.name,
                                        PFID(ll_inode2fid((*dentryp)->d_inode)),
                                        PFID(ll_inode2fid(inode)));
                                GOTO(out, rc = -ESTALE);
                        }

                        if ((bits & MDS_INODELOCK_LOOKUP) &&

                                d_lustre_revalidate(*dentryp);
