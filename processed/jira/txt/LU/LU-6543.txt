--- Description ---
lfs_setstripe() returns on an error return without free'ing param

                if (st_count > 0 && nr_osts != st_count) 

{
                        fprintf(stderr, "error: %s: stripe count '%d' doesn't "
                                "match the number of OSTs: %d\n",
                                argv[0], st_count, nr_osts);
                        return CMD_HELP;
                }

should be:

                if (st_count > 0 && nr_osts != st_count) 

{
                        fprintf(stderr, "error: %s: stripe count '%d' doesn't "
                                "match the number of OSTs: %d\n",
                                argv[0], st_count, nr_osts);
                        free(param);
                        return CMD_HELP;
                }
