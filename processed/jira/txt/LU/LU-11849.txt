--- Description ---
Somehow lfs find ! --projid  didn't work well and it is very easy to reproduce this:

[root@exa4-1 client]# lfs find --projid 1002 foor

foor/not-project

[root@exa4-1 client]# lfs find ! --projid 1002 foor

foor

foor/2.file

foor/not-project      -------------> this should not be outputted here.

foor/4.file

foor/3.file

foor/5.file

foor/1.file

Behavior I would expect would be something like the '--ost' case:

[root@exa4-1 client]# lfs find ! --ost 0 foor

foor/3.file

foor/5.file

foor/1.file

[root@exa4-1 client]# lfs find  --ost 0 foor

foor/2.file

foor/not-project

foor/4.file
