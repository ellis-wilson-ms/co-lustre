#
# Set the device of a given facet as "read-only".
#
set_dev_readonly() {
        local facet=$1
        local svc=${facet}_svc

        if [[ $(facet_fstype $facet) = zfs ]] ||
           ! dm_flakey_supported $facet; then
                do_facet $facet $LCTL --device ${!svc} readonly
        else
                dm_set_dev_readonly $facet
        fi
}
