check_network() {
        local host=$1
        local max=$2
        local sleep=${3:-5}

        [ "$host" = "$HOSTNAME" ] && return 0

        if ! wait_for_function --quiet "ping -c 1 -w 3 $host" $max $sleep; then
                echo "$(date +'%H:%M:%S (%s)') waited for $host network ${max}s"
                exit 1
        fi
}
