--- Description ---
In the following example, lnetctl import won't give the right result:

# cat lnet.yaml
net:
-     net type: tcp6
      local NI(s):
      -     nid: 2001:db8:1234:abcd::10@tcp6
            interfaces:
                  0: eth1
-     net type: tcp4
      local NI(s):
      -     nid: 10.0.227.5@tcp4
            interfaces:
                  0: eth1

# lnetctl import lnet.yaml
# lnetctl net show
net:
-     net type: lo
      local NI(s):
      -     nid: 0@lo
            status: up
-     net type: tcp6
      local NI(s):
      -     nid: 2001:db8:1234:abcd::10@tcp6
            status: up
            interfaces:
                  0: eth1
-     net type: tcp4
      local NI(s):
      -     nid: 2001:db8:1234:abcd::10@tcp4
            status: up
            interfaces:
                  0: eth1
