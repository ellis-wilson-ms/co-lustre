@@ -621,7 +621,6 @@ load_modules_local() {
     load_module fid/fid
     load_module lmv/lmv
     load_module mdc/mdc
-    load_module osc/osc
     load_module lov/lov
     load_module mgc/mgc
     load_module obdecho/obdecho
@@ -656,6 +655,7 @@ load_modules_local() {
                load_module osp/osp
        fi
 
+       load_module osc/osc
        load_module llite/lustre
        [ -d /r ] && OGDB=${OGDB:-"/r/tmp"}
        OGDB=${OGDB:-$TMP}
