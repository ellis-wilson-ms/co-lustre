--- Comment (zam, Wed, 28 May 2025 17:27:34 +0000) ---
The root cause is that lfs migrate tries to convert the dir in cb_migrate_mdt_fini():

/* dir migration finished, shrink its stripes */
static int cb_migrate_mdt_fini(char *path, int p, int *dp, void *data,
                               struct dirent64 *de)
{
        struct find_param *param = data;
        struct lmv_user_md *lmu = param->fp_lmv_md;
        int lmulen = lmv_user_md_size(lmu->lum_stripe_count, lmu->lum_magic);
        int ret = 0;

        if (de && de->d_type != DT_DIR)
                goto out;

        if (*dp != -1) {
                /*
                 * close it before setxattr because the latter may destroy the
                 * original object, and cause close fail.
                 */
                ret = close(*dp);
                *dp = -1;
                if (ret)
                        goto out;
        }

        ret = setxattr(path, XATTR_NAME_LMV, lmu, lmulen, 0);
        if (ret == -EALREADY)
                ret = 0;
...

by invoking setxattr() syscall for a xattr in "trusted" namespace.  The syscall fails because Linux doesn't allow a non-root user (user w/o CAP_SYS_ADMIN) to access any xattr from the "trusted" namespace:

static int
xattr_permission(struct inode *inode, const char *name, int mask)
{
   ...
        /*
         * The trusted.* namespace can only be accessed by privileged users.
         */
        if (!strncmp(name, XATTR_TRUSTED_PREFIX, XATTR_TRUSTED_PREFIX_LEN)) {
                if (!capable(CAP_SYS_ADMIN))
                        return (mask & MAY_WRITE) ? -EPERM : -ENODATA;
                return 0;
        }
...

moreover, it fails silently, as the error from the setxattr() call is ignored.
