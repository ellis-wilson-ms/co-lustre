--- Description ---
A user-space application using Lustre and GLib would face build issues, making them incompatible.

foo.c:

#include 
#include 

 

$ gcc -I/usr/include/glib-2.0 -I/usr/lib/x86_64-linux-gnu/glib-2.0/include -c foo.c 
In file included from /usr/lib/x86_64-linux-gnu/glib-2.0/include/glibconfig.h:9,
                 from /usr/include/glib-2.0/glib/gtypes.h:34,
                 from /usr/include/glib-2.0/glib/galloca.h:34,
                 from /usr/include/glib-2.0/glib.h:32,
                 from foo.c:4:
/usr/include/glib-2.0/glib/gmacros.h:682:28: error: missing ')' after "__has_attribute"
  682 | #if g_macro__has_attribute(fallthrough)
      |                            ^~~~~~~~~~~
In file included from foo.c:2:
/usr/include/glib-2.0/glib/gmacros.h:682:28: error: missing binary operator before token "("
  682 | #if g_macro__has_attribute(fallthrough)
      |                            ^~~~~~~~~~~

This is because lustreapi.h redefines the fallthrough attribute:

 

lustreapi.h:

#ifndef fallthrough
# if defined(__GNUC__) && __GNUC__ >= 7
#  define fallthrough  __attribute__((fallthrough)) /* fallthrough */
# else
#  define fallthrough do {} while (0)  /* fallthrough */
# endif
#endif

glib:

#if g_macro__has_attribute(fallthrough)
#define G_GNUC_FALLTHROUGH __attribute__((fallthrough)) \
  GLIB_AVAILABLE_MACRO_IN_2_60
#else
#define G_GNUC_FALLTHROUGH \
  GLIB_AVAILABLE_MACRO_IN_2_60
#endif

 

This is preventing to build some user-space tools like Robinhood, which is using both libraries.

	gcc: 13.3

	libglib2.0: 2.80
