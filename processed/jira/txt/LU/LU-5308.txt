                 struct obd_device *obd)
{
        struct ofd_thread_info  *info = ofd_info(env);
        struct dt_object        *fo;
        int                      rc = 0;

        ENTRY;

        rc = ofd_seqs_init(env, ofd);
        if (rc)
                GOTO(out_hc, rc);

         ....

        ofd->ofd_health_check_file = fo;

        RETURN(0);
out_hc:
        lu_object_put(env, &ofd->ofd_health_check_file->do_lu);
out:
        return rc;
}
