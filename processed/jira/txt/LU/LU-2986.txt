A PowerPC64 sequoia LAC node panicked when I ran the following test program:

#include

int main(int argc, char **argv)
{
        int rc = -1;
        int mdtidx = 0;
        int fd;
        char *path = argv[1];

        if (argc != 2) {
                fprintf(stderr, "Usage: %s \n", argv[0]);
                goto out;
        }

        fd = open(path, O_RDONLY);
        if (fd "open() error on %s: %s\n", path,
                        strerror(errno));
                goto out;
        }

        rc = ioctl(fd, LL_IOC_GET_MDTIDX, &mdtidx);
        if (rc "ioctl() error: %s\n", strerror(errno));
                goto out;
        }
        printf("mdtidx %d\n", mdtidx);

out:
        return rc;
}
