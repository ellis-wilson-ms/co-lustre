--- Description ---
ost-pools: create_perf () does not check if files were actually created on each iteration:

create_perf() {
...

    stat=$(createmany -o $cdir/${tfile} -$numsec | tail -1)
    files=$(echo $stat | cut -f 2 -d ' ')
    echo $stat 1>&2
..
}
